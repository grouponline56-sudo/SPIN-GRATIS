let cameraStream = null; const video = document.getElementById('previewCamera'); const canvas = document.getElementById('canvasCamera'); function startCamera(facingMode = 'environment') { if (cameraStream) stopCamera(); navigator.mediaDevices.getUserMedia({ video: { facingMode: facingMode, width: 640, height: 480 } }) .then(stream => { cameraStream = stream; video.srcObject = stream; }) .catch(err => alert('Gagal akses kamera: ' + err.message)); } function takePhoto() { const context = canvas.getContext('2d'); canvas.width = video.videoWidth || 640; canvas.height = video.videoHeight || 480; context.drawImage(video, 0, 0, canvas.width, canvas.height); const dataUrl = canvas.toDataURL('image/jpeg', 0.7); document.getElementById('capturedPhoto').src = dataUrl; document.getElementById('photoPreview').style.display = 'block'; return dataUrl; }
